 91        self.constraints.add_rule(conditioned, conditions, operator)

 94    def fit(self, epoch: int, objective=None, callback=None):
 95        """
 96        This is the fit method, call it to start the training process of the MetaHeuristic.
 97        :param epoch: Number of epoch to run the MetaHeuristic.
 98        :param objective: Objective fitness value, the fit will stop when an individual reaches this value.
 99        Select 'None' to ignore this feature.
100        :param callback: A callback function that is called at the end of the epoch with the following format:\n
101        def my_callback(epoch_number, population: tf.Tensor, score: tf.Tensor) -> bool:\n
102            [...] \n
103            return True \n
104        The population is a MxN shape matrix, and the score is an M dimensional vector. (M is the number of individuals
105        and N is the number of the parameters).\n
106        If it returns True, it allows the fit method to keep working. If False, it stops the fitting method.
107        :return: The trained population, the fitness score according to the population.
108    """
109        with ComputationalScope(self.ray[0], self.ray[1]) as cscope:
110            self._cscope = cscope
111            with Dashboard(self.constraints, self.extra_plots, cscope, cnt=self.each_dashboard) as dashboard:
112                #
113                #   Initialization.
114                #
115                self.population = self._initializer_(self.number_of_individuals, self.constraints)
116                self.population = self.constraints.apply_bindings(self.population)
117                self.population = self._initialization_correction(self.population)
118                new_individuals = self.population
119                self.identification = tf.cast(tf.linspace(0, new_individuals.shape[0],
120                                                          new_individuals.shape[0] + 1), tf.int32)
121                performance = {'fitness': 0, 'crossover': 0, 'selection': 0}
122                for epoch_number in range(epoch):
123                    tik = time.perf_counter()
124                    #
125                    #   Get the fitness function and sort in descending order.
126                    #
127                    divs = self._divide(new_individuals, self.ray[2])   # Divide into computational segments.
128                    cscope.bind(self.fitness)                           # The fitness function will be run.
129                    cscope.run(divs)                                    # Run the computational segments.
130                    futures = self._collapse(cscope.get())              # Obtain and collapse the results.
131                    self.score = self._idiot_proof_futures(futures)   # Some idiot proofing of the fitness function.
132                    self.population, self.score = self._sort()      # Sort the individuals according to their score.
133                    tak = time.perf_counter()                           # Tik-toc.
134                    performance['fitness'] = tak - tik                  # Performance store.
135                    #
136                    #   Additional actions per epoch:
137                    #
138                    #   Callback.
139                    #
140                    if callback is not None:
141                        if not callback(epoch_number, self.population, self.score):
142                            break
143                    #   Dashboard.
144                    #
145                    if self.there_is_dashboard:
146                        dashboard.add(self.score, self.population[:, :600], performance)
147                        dashboard.refresh_tab()
148                    #      Objective check.
149                    #
150                    if objective is not None:
151                        if max(self.score) >= objective:
152                            logging.info(f'The objective {objective} was achieved in epoch {epoch_number}.')
153                            break
154                    #
155                    #   Crossover.
156                    #
157                    # divs = (round(self.new_individuals_per_epoch / self.ray[2]), self.population)
158                    # cscope.bind(self._crossover_)                       # The crossover function will be run.
159                    # cscope.run(*divs)                                   # Run the computational segments.
160                    # futures = self._collapse(cscope.get())              # Obtain and collapse the results.
161                    # new_individuals = self.constraints.apply_bindings(futures)      # New pop.
162                    tak = time.perf_counter()                             # Tik-toc.
163                    new_individuals = self.constraints.apply_bindings(self._crossover_(self.population))  # New pop.
164                    new_individuals = self._crossover_correction(new_individuals)   # Correct the wrong individuals.
165                    tik = time.perf_counter()                            # Tik-toc.
166                    performance['crossover'] = tik - tak                 # Performance store.
167                    #
168                    #   Selection.
169                    #
170                    self.population = self._selection_(self.population, new_individuals)    # Selection in the pop.
171                    tak = time.perf_counter()                                               # Tik-tok.
172                    performance['selection'] = tak - tik                                    # Performance store.
173        return self.population, self.score

193    def _initialization_correction(self, new_individuals):
194        disrespectful = self.constraints.check_constraints(new_individuals)  # Rules.
195        if disrespectful:
196
197            correct_list = list()
198            if not self.__idiot_proof_told[0]:
199                logging.warning(f'Idiot-Proof-Warning: Your initializer does not respect the constraints of the problem'
200                                f'. Check your rules and your initializer.\nAutocorrection is enabled, '
201                                f'it will decrease the performance of your MetaHeuristic.')
202                self.__idiot_proof_told[0] = True
203            with alive_bar(self.number_of_individuals, title='Correcting initialization: ', manual=True) as bar:
204                while len(correct_list) < self.number_of_individuals:               # While the rules are not respected:
205                    correct = self._append_respectful(disrespectful, new_individuals)   # Get correct individuals.
206                    correct_list.extend(correct)                                            # Extend the correct list.
207                    new = self._initializer_(self.number_of_individuals, self.constraints)  # New individuals.
208                    new_individuals = self.constraints.apply_bindings(new)          # Apply constraints (min, max, type)
209                    disrespectful = self.constraints.check_constraints(new_individuals)     # Who does not respect?
210                    bar(len(correct[:self.number_of_individuals]) / self.number_of_individuals)
211                bar(1)
212            return tf.convert_to_tensor(correct_list[:self.number_of_individuals])
213        else:
214            return new_individuals

216    def _crossover_correction(self, new_individuals):
217        disrespectful = self.constraints.check_constraints(new_individuals)  # Rules.
218        if disrespectful:
219            if not self.__idiot_proof_told[1]:
220                logging.warning(f'Idiot-Proof-Warning: Your crossover does not respect the constraints of the problem'
221                                f'. Check your rules and your crossover.\nAutocorrection is enabled, '
222                                f'it will decrease the performance of your MetaHeuristic.')
223                self.__idiot_proof_told[1] = True
224            correct_list = list()
225            with alive_bar(self.new_individuals_per_epoch, title='Correcting crossover: ', manual=True) as bar:
226                while len(correct_list) < self.new_individuals_per_epoch:           # While the rules are not respected:
227                    correct = self._append_respectful(disrespectful, new_individuals)   # Get the correct individuals.
228                    correct_list.extend(correct)                                        # Extend the correct list.
229                    new = self._crossover_(self.population)  # New individuals.
230                    new_individuals = self.constraints.apply_bindings(new)          # Apply constraints (min, max, type)
231                    disrespectful = self.constraints.check_constraints(new_individuals)  # Who does not respect?
232                    bar(len(correct[:self.new_individuals_per_epoch]) / self.new_individuals_per_epoch)
233                bar(1)
234            return tf.convert_to_tensor(correct_list[:self.new_individuals_per_epoch])
235        else:
236            return new_individuals

238    @staticmethod
239    def _append_respectful(drp, ni):
240        totals = set(range(len(ni)))
241        correct = list()
242        for element in totals:
243            if element not in drp:
244                correct.append(ni[element])
245        return correct

247    def _sort(self) -> tuple[tf.Tensor, tf.Tensor]:
248        indices = tf.argsort(self.score, direction='DESCENDING')
249        pop = tf.gather(self.population, indices)
250        self.identification = tf.gather(self.identification, indices)
251        return pop, tf.sort(self.score, direction='DESCENDING')

253    @staticmethod
254    def _divide(population: tf.Tensor, divisions: int):
255        individuals_per_division = round(len(population) / divisions)
256        divs = list()
257        for index in range(divisions - 1):
258            divs.append(population[index * individuals_per_division:(index + 1) * individuals_per_division].numpy())
259        divs.append(population[(divisions - 1) * individuals_per_division:].numpy())
260        return divs

262    def _collapse(self, futures):
263        collapsed = list()
264        for future in futures:
265            collapsed.extend(future)
266        news = tf.convert_to_tensor(collapsed)
267        if self.score is None:
268            return news
269        return tf.concat([self.score[0:-self.new_individuals_per_epoch], news], 0)

273    def _initializer_(self, number_of_individuals, constraints):
274        try:
275            _return_value_ = self.initializer(number_of_individuals, constraints)
276            if len(_return_value_) == self.number_of_individuals:
277                if len(_return_value_[0]) != len(self.constraints.parameters):
278                    logging.warning(f'Idiot-Proof-Warning: Your initializer does not return '
279                                    f'{len(self.constraints.parameters)} parameters per individual, it returns '
280                                    f'{len(_return_value_[0])} parameters per individual. Fix your initializer.')
281            else:
282                logging.warning(f'Idiot-Proof-Warning: Your initializer does not return {number_of_individuals} '
283                                f'individuals, it returns {len(_return_value_)} individuals. Fix your initializer.')
284            if isinstance(_return_value_, tf.Tensor):
285                return _return_value_
286            else:
287                logging.warning(f'Idiot-Proof-Comment: Your initializer does not return a tf.Tensor as a population. It'
288                                f' returns a "{type(_return_value_)}" type. I am trying to convert it to "tf.Tensor".')
289                return tf.convert_to_tensor(_return_value_)
290        except Exception as ex:
291            logging.error(f'An exception raised from the initializer method: {ex}.')
292            raise ex

341        _return_value_ = tf.convert_to_tensor(futures_values)
342        if len(_return_value_) != len(self.population):
343            logging.warning(f'Idiot-Proof-Warning: Your fitness method does not return '
